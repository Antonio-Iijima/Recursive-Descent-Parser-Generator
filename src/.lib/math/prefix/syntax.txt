<MATHEXPR> ::= <ADD> | <SUBTRACT> | <TERM>
<ADD>      ::= <OP_ADD> <MATHEXPR> <TERM>
<SUBTRACT> ::= <OP_SUBTRACT> <MATHEXPR> <TERM>
<TERM>     ::= <MULTIPLY> | <DIVIDE> | <FACTOR>
<MULTIPLY> ::= <OP_MULTIPLY> <TERM> <FACTOR>
<DIVIDE>   ::= <IDIVIDE> | <FDIVIDE>
<FACTOR>   ::= <LPAREN> <MATHEXPR> <RPAREN> | <ABS> | <EXP> | <NUMBER> | <OP_SUBTRACT> <FACTOR>
<ABS>      ::= <OP_PIPE> <MATHEXPR> <OP_PIPE>
<IDIVIDE>  ::= <OP_IDIVIDE> <TERM> <FACTOR>
<FDIVIDE>  ::= <OP_FDIVIDE> <TERM> <FACTOR>
<EXP>      ::= <OP_EXP> <FACTOR> <FACTOR>
