<MATHEXPR> ::= <ADD> | <SUBTRACT> | <TERM>
<ADD>      ::= <MATHEXPR> <OP_ADD> <TERM>
<SUBTRACT> ::= <MATHEXPR> <OP_SUBTRACT> <TERM>
<TERM>     ::= <MULTIPLY> | <DIVIDE> | <FACTOR>
<MULTIPLY> ::= <TERM> <OP_MULTIPLY> <FACTOR>
<DIVIDE>   ::= <IDIVIDE> | <FDIVIDE>
<FACTOR>   ::= <LPAREN> <MATHEXPR> <RPAREN> | <ABS> | <EXP> | <NUMBER> | <OP_SUBTRACT> <FACTOR>
<ABS>      ::= <OP_PIPE> <MATHEXPR> <OP_PIPE>
<IDIVIDE>  ::= <TERM> <OP_IDIVIDE> <FACTOR>
<FDIVIDE>  ::= <TERM> <OP_FDIVIDE> <FACTOR>
<EXP>      ::= <FACTOR> <OP_EXP> <FACTOR>
