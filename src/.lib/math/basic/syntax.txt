<MATHEXPR>   ::= <ADD> | <SUBTRACT> | <TERM>
<ADD>        ::= <MATHEXPR> <OP_ADD> <TERM>
<SUBTRACT>   ::= <MATHEXPR> <OP_SUBTRACT> <TERM>
<TERM>       ::= <MULTIPLY> | <DIVIDE> | <OP_SUBTRACT> <TERM> | <MODULO> | <FACTOR>
<MULTIPLY>   ::= <TERM> <OP_MULTIPLY> <FACTOR>
<DIVIDE>     ::= <TERM> <OP_FDIVIDE> <FACTOR>
<MODULO>     ::= <TERM> <OP_MODULO> <FACTOR>
<FACTOR>     ::= <LPAREN> <MATHEXPR> <RPAREN> | <COMPARISON> | <NUMBER>

<COMPARISON> ::= <LE> | <LEQ> | <GE> | <GEQ> | <EQ> | <NEQ>
<LE>         ::= <MATHEXPR> <OP_LE> <MATHEXPR>
<LEQ>        ::= <MATHEXPR> <OP_LEQ> <MATHEXPR>
<GE>         ::= <MATHEXPR> <OP_GE> <MATHEXPR>
<GEQ>        ::= <MATHEXPR> <OP_GEQ> <MATHEXPR>
<EQ>         ::= <MATHEXPR> <OP_EQ> <MATHEXPR>
<NEQ>        ::= <MATHEXPR> <OP_NEQ> <MATHEXPR>
